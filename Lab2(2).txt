#include <iostream>
using namespace std;

int main()
{
	string str;
	cout << "Enter a string: ";
	cin >> str;
	int c = 1;
	char Operator[6] = { '+','-','*','/','%','='};
	for (int i = 0; i <str.length(); i++)
	{
		for(int j=0;j<6;j++)
		{
			if (str[i] == Operator[j])
			{
				cout << "Operator " << c++ << " : " << str[i] << endl;
				break;
			}
		}
	}


}
